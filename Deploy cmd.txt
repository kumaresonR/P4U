1. CONNECT TO SERVER
ssh root@YOUR_SERVER_IP

2. INITIAL SETUP
passwd
apt update
apt upgrade -y
reboot

3. FIREWALL SETUP
apt install -y ufw
ufw allow OpenSSH
ufw allow 80/tcp
ufw allow 443/tcp
ufw enable
ufw status

4. INSTALL BASIC TOOLS
apt install -y git curl build-essential

5. INSTALL NODEJS (v20)
curl -fsSL https://deb.nodesource.com/setup_20.x | bash -
apt install -y nodejs
node -v
npm -v

6. INSTALL PM2
npm install -g pm2
pm2 -v

7. INSTALL POSTGRESQL
apt install -y postgresql postgresql-contrib
systemctl status postgresql

8. CREATE DATABASE
sudo -u postgres psql
CREATE USER planext WITH PASSWORD 'StrongPassword123!';
CREATE DATABASE planext4u OWNER planext;
\q

9. INSTALL REDIS
apt install -y redis-server
systemctl enable redis-server
systemctl start redis-server
redis-cli ping   → should return PONG

10. CLONE PROJECT
cd /var/www
git clone https://github.com/YOUR_REPO.git
cd YOUR_REPO/planext4u-backend

11. ENV SETUP
cp .env.example .env
nano .env

ADD:
NODE_ENV=production
PORT=5000
DATABASE_URL=postgresql://planext:StrongPassword123!@localhost:5432/planext4u
REDIS_URL=redis://127.0.0.1:6379
JWT_ACCESS_SECRET=your_secret
JWT_REFRESH_SECRET=your_secret

SAVE:
CTRL + X → Y → ENTER

12. INSTALL DEPENDENCIES
npm ci

13. PRISMA SETUP
npx prisma generate
npx prisma migrate deploy

npm run build

pm2 start dist/server.js --name backend
